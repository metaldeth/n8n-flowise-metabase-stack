# ========================
# HTTP (порт 80)
# ========================
server {
    listen 80;
    server_name n8n.asnefedov.ru flowise.asnefedov.ru metabase.asnefedov.ru pg.asnefedov.ru;

    # Challenge для Let's Encrypt
    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    # Все остальные запросы переводим на HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}

# ========================
# HTTPS (порт 443)
# ========================

# n8n
server {
    listen 443 ssl http2;
    server_name n8n.asnefedov.ru;

    ssl_certificate     /etc/letsencrypt/live/n8n.asnefedov.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/n8n.asnefedov.ru/privkey.pem;

    location / {
        proxy_pass http://n8n:5678;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
